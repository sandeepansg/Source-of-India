% Article mapping for Indian Constitution
% Maps logical article numbers to actual file names

\newcommand{\includeArticle}[1]{%
    \IfFileExists{includes/articles/article-#1.tex}{%
        \input{includes/articles/article-#1.tex}%
    }{%
        \PackageWarning{soi}{Article #1 not found - check amendment status}%
    }%
}

% Include articles by Part
\newcommand{\includePartArticles}[2]{%
    \forloop{articleloop}{#1}{\value{articleloop} <= #2}{%
        \includeArticle{\arabic{articleloop}}%
    }%
}

% Check if article exists
\newcommand{\ifArticleExists}[3]{%
    \IfFileExists{includes/articles/article-#1.tex}{#2}{#3}%
}
